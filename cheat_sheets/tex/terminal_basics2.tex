        \begin{tabular}{rl}
            \lc{find .\ -name "*.txt"}{find all files with a ".txt" ending}
            \lc{find .\ -type d}{find all directories in the current folder}
            \lc{find .\ -name "c*" -type d}{find all directories starting with a "c"}
